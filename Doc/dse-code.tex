const int THRESHOLD = 100;
int logicalValue(int t) {
  int elapsed = currentTime - t;
  int val = 0;
  if (elapsed < THRESHOLD) { 
    val = 1; 
  } else {
    for (int i = 0 ; i < data.length; i++)

    old = val;      
  }
  return val;
}
