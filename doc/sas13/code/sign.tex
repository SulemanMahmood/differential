int sign(int x) {
  int sgn;

    sgn = -1
  else
    sgn = 1
 return sgn

int sign(int x) {
  int x' = x;
  guard g1 = (x < 0);
  guard g1' = (x' < 0);
  int sgn;
  int sgn' = sgn;
  if (g1) sgn = -1;
  if (g1') sgn' = -1;
  if (!g1) sgn = 1;
  if (!g1') sgn' = 1;
  guard g2' = (x' == 0);
  if (g2') sgn' = 0;
}
\end{lstlisting}
